   using System;
using System.Text;

namespace ConsoleApp4
{
    class Program
    {
        static void Main(string[] args)
        {
            int num = 35;
            Console.WriteLine(similar(ref num));
            Console.WriteLine(num);
        }

        static int similar(ref int num)
        {

            int a = 0;
            int i = 0;
            for (i = 0; i*i <= num; i++)
            {
                a= i;
            }
            

            if (num-a*a>=i*i-num)
            {
                    num = i;
             }
            else
                {
                    num = a;
                }



            return num;



        }
    }
}
