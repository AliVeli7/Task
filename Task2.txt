   using System;
using System.Text;

namespace ConsoleApp4
{
    class Program
    {
        static void Main(string[] args)
        {


            string str = "Hello How are you?";
            Console.WriteLine(delSpace(ref str));
            Console.WriteLine(str);
        }

        static string delSpace(ref string str)
        {

            
           
                for (int i = 0; i < str.Length; i++)
                {
                   
                    if (str[i] == ' ')
                    {
                   
                   str = str.Remove(i,1);

                }
                else
                {
                    continue;

                }
                }

            return str;
        }
    }
}
